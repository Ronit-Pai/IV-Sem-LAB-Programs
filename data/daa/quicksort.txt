#include <stdio.h>
int partition(int a[], int l, int r)
{
    int pivot = a[l];
    int i = l+1;
    int j = r;
    while (i <= j)
    {
        while (i <= r && a[i] <= pivot)
            i++;

        while (j >= l && a[j] > pivot)
            j--;

        if (i < j)
        {
            int temp = a[i];
            a[i] = a[j];
            a[j] = temp;
            i++;
            j--;
        }
    }
    a[l] = a[j];
    a[j] = pivot;
    return j;
}
void quicksort(int a[],int l,int r)
{
    if (l<r)
    {
        int s = partition(a, l, r);
        quicksort(a,l,s-1);
        quicksort(a,s+1,r);
    }
}
int main()
{
    int n;
    printf("Enter the size of array: ");
    scanf("%d", &n);
    int a[n];
    printf("Enter the array elements:\n");
    for (int i = 0; i < n; i++)
        scanf("%d", &a[i]);
    int l = 0, r = n - 1;
    quicksort(a,l,r);
    printf("Sorted Array:\n");
    for (int i = 0; i < n; i++)
        printf("%d \t",a[i]);
    return 0;
}
